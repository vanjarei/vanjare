Data Manipulation Language (DML)

SELECT


SELECT FirstName, LastName FROM Employees WHERE HireDate > '2023-01-01';

INSERT


INSERT INTO Employees (EmployeeID, FirstName, LastName, Email, HireDate)
VALUES (1, 'John', 'Doe', 'john.doe@example.com', '2024-01-10');

UPDATE


UPDATE Employees SET Email = 'updated.email@example.com' WHERE EmployeeID = 1;

DELETE


DELETE FROM Employees WHERE EmployeeID = 1;

MERGE


MERGE INTO Employees AS Target
USING (SELECT 1 AS EmployeeID, 'Jane' AS FirstName, 'Smith' AS LastName) AS Source
ON Target.EmployeeID = Source.EmployeeID
WHEN MATCHED THEN
    UPDATE SET FirstName = Source.FirstName, LastName = Source.LastName
WHEN NOT MATCHED THEN
    INSERT (EmployeeID, FirstName, LastName)
    VALUES (Source.EmployeeID, Source.FirstName, Source.LastName);

Transaction Control Language (TCL)

BEGIN TRANSACTION


BEGIN TRANSACTION;
INSERT INTO Employees (EmployeeID, FirstName, LastName) VALUES (2, 'Alice', 'Johnson');
COMMIT;

ROLLBACK


BEGIN TRANSACTION;
INSERT INTO Employees (EmployeeID, FirstName, LastName) VALUES (3, 'Bob', 'Brown');
ROLLBACK;

SAVE TRANSACTION


BEGIN TRANSACTION;
INSERT INTO Employees (EmployeeID, FirstName, LastName) VALUES (4, 'Charlie', 'Clark');
SAVE TRANSACTION SavePoint1;
DELETE FROM Employees WHERE EmployeeID = 4;
ROLLBACK TRANSACTION SavePoint1; -- Reverts to the save point
COMMIT;
